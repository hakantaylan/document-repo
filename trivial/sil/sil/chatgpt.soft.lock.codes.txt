1Ô∏è‚É£ Use Optimistic Locking (mandatory)

Add a version column to the entity. This is the primary mechanism that prevents lost updates.

When the user reads the record ‚Üí they get version = X

When they update ‚Üí Hibernate generates:

SET ..., version = version + 1

4Ô∏è‚É£ Base Locking Repository Interface (What Services Use)
public interface LockingRepository {

    /**
     * Tries to acquire a lock.
     * @return true if lock acquired, false if locked by someone else
     */
    boolean tryLock(
        Class<?> entityClass,
        Long entityId,
        String username,
        Duration duration
    );

    void releaseLock(
        Class<?> entityClass,
        Long entityId,
        String username
    );
}

5Ô∏è‚É£ Base Locking Repository Implementation
@Repository
@Transactional
public class JpaLockingRepository implements LockingRepository {

    private final EntityLockRepository lockRepository;

    public JpaLockingRepository(EntityLockRepository lockRepository) {
        this.lockRepository = lockRepository;
    }

    @Override
    public boolean tryLock(
            Class<?> entityClass,
            Long entityId,
            String username,
            Duration duration
    ) {
        String entityType = entityClass.getSimpleName();
        LocalDateTime now = LocalDateTime.now();

        return lockRepository
            .findByEntityTypeAndEntityId(entityType, entityId)
            .map(existing -> {
                if (!existing.isExpired(now)) {
                    return false;
                }
                lockRepository.delete(existing);
                lockRepository.save(
                    new EntityLock(entityType, entityId, username, now.plus(duration))
                );
                return true;
            })
            .orElseGet(() -> {
                try {
                    lockRepository.save(
                        new EntityLock(entityType, entityId, username, now.plus(duration))
                    );
                    return true;
                } catch (DataIntegrityViolationException ex) {
                    // Unique constraint hit ‚Üí someone else locked concurrently
                    return false;
                }
            });
    }

    @Override
    public void releaseLock(
            Class<?> entityClass,
            Long entityId,
            String username
    ) {
        String entityType = entityClass.getSimpleName();

        lockRepository.findByEntityTypeAndEntityId(entityType, entityId)
            .filter(lock -> lock.getLockedBy().equals(username))
            .ifPresent(lockRepository::delete);
    }
}

4Ô∏è‚É£ Base Repository Implementation (Core Part)

This is where the magic happens.

public class LockingRepositoryImpl<T, ID>
        implements LockingRepository<ID> {

    @PersistenceContext
    private EntityManager em;

    private final JpaEntityInformation<T, ?> entityInfo;

    public LockingRepositoryImpl(
            JpaEntityInformation<T, ?> entityInfo,
            EntityManager em
    ) {
        this.entityInfo = entityInfo;
        this.em = em;
    }

    @Override
    @Transactional
    public boolean tryLock(ID id, String username, Duration duration) {

        String table = entityInfo.getTableName();
        LocalDateTime now = LocalDateTime.now();
        LocalDateTime until = now.plus(duration);

        String sql = """
            UPDATE %s
            SET locked_by = :user,
                lock_until = :until
            WHERE id = :id
              AND (
                    lock_until IS NULL
                 OR lock_until < :now
                 OR locked_by = :user
              )
            """.formatted(table);

        int updated = em.createNativeQuery(sql)
            .setParameter("id", id)
            .setParameter("user", username)
            .setParameter("now", now)
            .setParameter("until", until)
            .executeUpdate();

        return updated == 1;
    }

    @Override
    @Transactional
    public void releaseLock(ID id, String username) {

        String table = entityInfo.getTableName();

        String sql = """
            UPDATE %s
            SET locked_by = NULL,
                lock_until = NULL
            WHERE id = :id
              AND locked_by = :user
            """.formatted(table);

        em.createNativeQuery(sql)
            .setParameter("id", id)
            .setParameter("user", username)
            .executeUpdate();
    }
}

3Ô∏è‚É£ Base Repository Implementation
public class LockAwareRepositoryImpl<T, ID>
        implements LockAwareRepository<ID> {

    @PersistenceContext
    private EntityManager em;

    private final JpaEntityInformation<T, ?> entityInfo;

    public LockAwareRepositoryImpl(
            JpaEntityInformation<T, ?> entityInfo,
            EntityManager em
    ) {
        this.entityInfo = entityInfo;
        this.em = em;
    }

    @Override
    @Transactional
    public boolean tryLock(ID id, String user, Duration duration) {

        String table = entityInfo.getTableName();
        LocalDateTime now = LocalDateTime.now();

        String sql = """
            UPDATE %s
               SET locked_by = :user,
                   lock_until = :until
             WHERE id = :id
               AND (
                    lock_until IS NULL
                 OR lock_until < :now
                 OR locked_by = :user
               )
            """.formatted(table);

        int updated = em.createNativeQuery(sql)
            .setParameter("id", id)
            .setParameter("user", user)
            .setParameter("now", now)
            .setParameter("until", now.plus(duration))
            .executeUpdate();

        return updated == 1;
    }

    @Override
    @Transactional
    public void releaseLock(ID id, String user) {

        String table = entityInfo.getTableName();

        em.createNativeQuery("""
            UPDATE %s
               SET locked_by = NULL,
                   lock_until = NULL
             WHERE id = :id
               AND locked_by = :user
            """.formatted(table))
            .setParameter("id", id)
            .setParameter("user", user)
            .executeUpdate();
    }

    @Override
    @Transactional
    public void assertValidLock(ID id, String user) {

        String table = entityInfo.getTableName();

        int valid = ((Number) em.createNativeQuery("""
            SELECT COUNT(*)
              FROM %s
             WHERE id = :id
               AND locked_by = :user
               AND lock_until > :now
            """.formatted(table))
            .setParameter("id", id)
            .setParameter("user", user)
            .setParameter("now", LocalDateTime.now())
            .getSingleResult()).intValue();

        if (valid == 0) {
            throw new LockExpiredException(
                "Lock expired or not owned by user"
            );
        }
    }
}

    private final JpaEntityInformation<T, ?> entityInfo;

            EntityManager em) {
        super(entityInfo, em);
        this.em = em;
        this.entityInfo = entityInfo;
    }

    @Override

        LocalDateTime now = LocalDateTime.now();
        int updated = em.createQuery("""

               SET e.lockUser = :user,

        .executeUpdate();
        return updated == 1;
    }

    @Override
    @Transactional

        em.createQuery("""
            UPDATE %s e
               SET e.lockUser = NULL,
                   e.lockExpireDate = NULL
             WHERE e.id = :id
               AND e.lockUser = :user
        """.formatted(entityInfo.getEntityName()))
        .setParameter("id", id)
        .setParameter("user", user)
        .executeUpdate();
    }

    // --- Override mutating methods ---

    @Override
    @Transactional
    public <S extends T> S save(S entity) {
        validateLock(entity);
        return super.save(entity);
    }

    @Override
    @Transactional
    public <S extends T> List<S> saveAll(Iterable<S> entities) {
        for (S e : entities) validateLock(e);
        return super.saveAll(entities);
    }

    @Override
    @Transactional
    public void delete(T entity) {
        validateLock(entity);
        super.delete(entity);
    }

    @Override
    @Transactional

        for (T e : entities) validateLock(e);
        super.deleteAll(entities);
    }

    // --- lock validation helper ---
    private void validateLock(T entity) {
        if (entity.lockExpireDate == null
            || entity.lockExpireDate.isBefore(LocalDateTime.now())) {

        || entity.lockExpireDate.isBefore(LocalDateTime.now())) {
        throw new LockExpiredException("Lock expired or not owned by current user");
    }
}

2Ô∏è‚É£ Application-Level Enhancements

Once your repository enforces the lock, you can consider application-level strategies to make it more robust.

2.1 Java Synchronization

You can use in-memory locking (synchronized, ReentrantLock) per entity id.

This is useful within a single JVM to prevent multiple threads from simultaneously modifying the same entity before the database lock check.

Example:

private final ConcurrentHashMap<Long, ReentrantLock> lockMap = new ConcurrentHashMap<>();

public void updateOrderWithThreadSafety(Long orderId, Runnable action) {
    ReentrantLock lock = lockMap.computeIfAbsent(orderId, id -> new ReentrantLock());
    lock.lock();
    try {
        action.run();
    } finally {
        lock.unlock();
        // optionally remove lock if not needed
        lockMap.remove(orderId, lock);
    }
}


‚úÖ Pros:

Prevents thread-level race conditions

Works well in a single JVM

‚ùå Cons:

Does not help across multiple JVMs (clustered environment)

Must still rely on DB locks for multi-node safety

2.2 Optional: Soft lock heartbeat / auto-renew

You can refresh lockExpireDate periodically if the client is active.

Prevents the ‚Äúnetwork freeze‚Äù scenario from immediately expiring a lock.

Example: schedule a task or use WebSocket/REST heartbeat.

2.3 Optional: In-memory cache of locked ids

Keep recently locked entity ids in memory (like a ConcurrentHashMap) for fast read before DB check.

Reduces unnecessary DB calls in high-throughput systems.

Always fallback to DB for actual enforcement.

2.4 Optional: Domain Events / Observer

Emit events when a lock expires, is stolen, or released.

Can trigger notifications to clients to refresh their UI.

3Ô∏è‚É£ Summary of Enhancements
Strategy	Scope	Benefit
DB-guarded repository (override save)	All JVMs	Required for correctness
Java ReentrantLock / synchronized	Single JVM	Prevents thread-level races before DB commit
Heartbeat / auto-renew	Client-JVM	Prevents accidental expiration during long processing
In-memory lock cache	Single JVM	Reduces DB hits
Domain events / notifications	Any	Improves UX for stale locks

DB lock enforcement is the only thing that guarantees correctness in multi-node environments.

Java synchronization is optional, but it can prevent wasted work in single JVMs.

Heartbeat / cache / events improve UX but are not correctness mechanisms.

4Ô∏è‚É£ Locking Repository Implementation
public class LockingRepositoryImpl<T extends LockableEntity, ID>
        extends SimpleJpaRepository<T, ID>
        implements LockingRepository<T, ID> {

    private final EntityManager em;
    private final JpaEntityInformation<T, ?> entityInfo;

    public LockingRepositoryImpl(
            JpaEntityInformation<T, ?> entityInfo,
            EntityManager em) {
        super(entityInfo, em);
        this.em = em;
        this.entityInfo = entityInfo;
    }

    // --- Lock methods ---
    @Override
    @Transactional
    public boolean tryLock(ID id, String user, Duration duration) {
        LocalDateTime now = LocalDateTime.now();
        int updated = em.createQuery("""
            UPDATE %s e
               SET e.lockUser = :user,
                   e.lockExpireDate = :until
             WHERE e.id = :id
               AND (
                    e.lockExpireDate IS NULL
                 OR e.lockExpireDate < :now
                 OR e.lockUser = :user
               )
        """.formatted(entityInfo.getEntityName()))
        .setParameter("id", id)
        .setParameter("user", user)
        .setParameter("now", now)
        .setParameter("until", now.plus(duration))
        .executeUpdate();
        return updated == 1;
    }

    @Override
    @Transactional
    public void releaseLock(ID id, String user) {
        em.createQuery("""
            UPDATE %s e
               SET e.lockUser = NULL,
                   e.lockExpireDate = NULL
             WHERE e.id = :id
               AND e.lockUser = :user
        """.formatted(entityInfo.getEntityName()))
        .setParameter("id", id)
        .setParameter("user", user)
        .executeUpdate();
    }

    // --- Override mutating methods ---
    @Override
    @Transactional
    public <S extends T> S save(S entity) {
        validateLock(entity);
        return super.save(entity);
    }

    @Override
    @Transactional
    public <S extends T> List<S> saveAll(Iterable<S> entities) {
        for (S e : entities) validateLock(e);
        return super.saveAll(entities);
    }

    @Override
    @Transactional
    public void delete(T entity) {
        validateLock(entity);
        super.delete(entity);
    }

    @Override
    @Transactional
    public void deleteAll(Iterable<? extends T> entities) {
        for (T e : entities) validateLock(e);
        super.deleteAll(entities);
    }

    private void validateLock(T entity) {
        String currentUser = getCurrentUser();
        if (!entity.isLockValid(currentUser)) {
            throw new LockExpiredException("Lock expired or not owned by current user");
        }
    }

    private String getCurrentUser() {
        // Example using Spring Security
        return SecurityContextHolder.getContext().getAuthentication().getName();
    }
}
